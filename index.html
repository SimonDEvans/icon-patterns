Import font icons

Import the font icons you want. In this example we are using Ionicons.com. 

<!-- Load the icon set that you will be using in the HEAD section -->
<link href="https://cdnjs.cloudflare.com/ajax/libs/ionicons/2.0.1/css/ionicons.css" rel="stylesheet" type="text/css" >

Load Dependencies 

<!-- Load this in your head section -->
<!-- IconPatterns import -->
<link rel="stylesheet" type="text/css" href="./dist/icon-patterns.css">
<!-- Load this in before closing the body tag -->
<!-- jQuery import -->
<script src="https://code.jquery.com/jquery-3.2.1.min.js"></script>
<!-- IconPatterns import -->
<script src="./dist/icon-patterns.min.js" ></script>

Setup DOM 

<div id="container"> We've created Icon Patterns </div>

Configure Icon Patterns 

$('#container').IconPatterns({
  color: '#FFFFFF',
  icons: [{
    name: 'ion-erlenmeyer-flask',
    count: 15,
    sizeVariation: [20, 50],
  },{
    name: 'ion-cube',
    count: 10,
    sizeVariation: [40, 80],
  }]
}).draw();
