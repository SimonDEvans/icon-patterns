Setup DOM 

<div id="container"> Content </div>

Initialize icon patterns 

$(document).ready(() => {
  const patternAnimation = $('#container').IconPatterns({
    color: '#FFFFFF',
    icons: [
      {name: 'ion-erlenmeyer-flask'},
      {name: 'ion-fireball'}
    ]
  }).draw();
});

Customize icons and animations 

$(document).ready(() => {
  const ANIMATIONS = $.IconPatterns.ANIMATIONS;
  const patternAnimation = $('#container').IconPatterns({
    color: '#FFFFFF',
    icons: [
    {
      // Required name
      name: 'ion-erlenmeyer-flask',
    }, {
      // You can optionally define the animation styles you want
      name: 'ion-fireball',
      animations: [ANIMATIONS.ROTATE_SLOW, ANIMATIONS.EXPAND]
    }, {
      // Or you can customize in depth
      name: 'ion-cube',
      count: 10,
      sizeVariation: [40, 80],
      animations: [ANIMATIONS.ROTATE, ANIMATIONS.EXPAND_FAST]
    }]
  }).draw();

  // Bind redraw event on resize
  window.onresize = () => patternAnimation.redraw();
});
